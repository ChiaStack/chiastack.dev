`@chiastack/ui` 是一個 React UI 元件庫，提供可重複使用的 UI 元件，特別專注於交易圖表（TradingView 風格）和錯誤處理功能。

## 安裝

<Tabs items={['npm', 'pnpm', 'yarn', 'bun']}>
  <Tab value="npm">
    ```bash
    npm install @chiastack/ui
    ```
  </Tab>
  <Tab value="pnpm">
    ```bash
    pnpm add @chiastack/ui
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn add @chiastack/ui
    ```
  </Tab>
  <Tab value="bun">
    ```bash
    bun add @chiastack/ui
    ```
  </Tab>
</Tabs>

## 基本用法

```tsx
import { Chart } from '@chiastack/ui/trading-chart/chart';
import { ErrorBoundary } from '@chiastack/ui/error-boundary';

function App() {
  return (
    <ErrorBoundary fallback={<div>發生錯誤</div>}>
      <Chart>
        {/* Chart components */}
      </Chart>
    </ErrorBoundary>
  );
}
```
